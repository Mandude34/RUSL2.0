<h2>Recommendations</h2>
<div id="recommend"></div>

<script>
let inventory = [];
let sales = [];

let recipes = {
  "chicken bowl": { chicken: 4, rice: 6, sauce: 2 }
};

function render(){
  inventoryList.innerHTML = inventory.map(i =>
    `<div class="card">${i.name}: ${i.stock}</div>`).join("");

  salesList.innerHTML = sales.map(s =>
    `<div class="card">${s.item} x ${s.qty}</div>`).join("");

  let usage = {};
  sales.forEach(s => {
    let r = recipes[s.item];
    if(r){
      for(let k in r){
        usage[k] = (usage[k] || 0) + r[k]*s.qty;
      }
    }
  });

  let out = "";
  for(let k in usage){
    let inv = inventory.find(i => i.name === k);
    let stock = inv ? inv.stock : 0;
    let need = usage[k] - stock;
    if(need > 0){
      out += `<div class="card">Order ${need} ${k}</div>`;
    }
  }
  recommend.innerHTML = out || "All good";
}

function addInventory(){
  inventory.push({name:invName.value.toLowerCase(), stock:+invStock.value});
  invName.value=""; invStock.value="";
  render();
}

function addSale(){
  sales.push({item:saleItem.value.toLowerCase(), qty:+saleQty.value});
  saleItem.value=""; saleQty.value="";
  render();
}

render();
</script>
